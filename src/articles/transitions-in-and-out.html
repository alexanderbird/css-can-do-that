        .page:target {
          animation: fade-in 1s;
        }

        .page:target ~ #home-page,
        .page:not(#home-page):not(:target) {
          animation: fade-out 1s;
        }

        @keyframes fade-in {
          from { opacity: 0; } 
          to { opacity: 1; }
        }

        @keyframes fade-out {
          from { opacity: 1; }
          to { opacity: 0; }
        }
